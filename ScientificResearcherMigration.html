def cleanyear(phdYear,earliest):
    if (phdYear==phdYear) & (earliest!=earliest):
        return phdYear-4
    else:
        return earliest

def cleanyearv2(phdYear,earliest,hasPhd):
    if (phdYear!=phdYear) & (earliest==earliest) & (hasPhd==True) :
        return earliest+4
    else:
        return phdYear
